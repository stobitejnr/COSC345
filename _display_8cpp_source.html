556auto Display::fast_print(const std::basic_string<char_type> &sss) -> void
557{
558 HANDLE const output_handle = GetStdHandle(STD_OUTPUT_HANDLE);
559 const auto char_count = static_cast<DWORD>(sss.length());
560 if constexpr (std::is_same_v<char_type, char>)
561 WriteConsoleA(output_handle, sss.c_str(), char_count, nullptr, nullptr);
562 else
